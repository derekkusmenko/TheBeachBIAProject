\begin{frame}{Project Objectives}

    \begin{itemize}
        \item \textbf{Economic Vitality:} Understand foot traffic in the Beaches to increase activity for local businesses.
        \item \textbf{Data:} Leveraging Toronto Open Data to justify BIA budgetary decisions.
    \end{itemize}
    
    \begin{block}{Specific Research Questions}
        \begin{enumerate}
            \item Can we segment the people coming to the Beaches based off their habits and where they come from?
            \item How long do pedestrians stay at the Beaches for?
            \item How do external variables (weather, holidays) probabilistically impact pedestrian flow?
            \item How these patterns may affect commercial vitality along the strip
        \end{enumerate}
    \end{block}
\end{frame}


\begin{frame}{Data}
    
    \textbf{Case for Open-Source} \\
    \small The Beach BIA deliberately avoided commercial mobile phone data, relying exclusively on municipal datasets for The Beach BIA.
    \vspace{0.1cm}
    \begin{itemize}
        \item \textbf{Privacy \& Trust:} Eliminates re-identification risks, vulnerable demographics (youth/seniors) inherently under-represented by mobile tracking.
        \item To avoid expensive purchasing of data, municipal datasets are purpose-built for urban planning and already tax-funded.
    \end{itemize}
    
    \vspace{0.4cm} % Adds breathing room between the sections
    
    \textbf{Datasets} \\
    \small We aggregated multiple open-source domains from \textbf{2021--2025} into a single analytical dataset.
    \vspace{0.1cm}
    \begin{itemize}
        \item Extracted Open Data Toronto Bike Share records, strictly filtering for trips originating or terminating at Beach BIA docks.
        \item Merged the core bike data with Historical Climate Data (Environment Canada), Toronto Beach Water Quality metrics.
    \end{itemize}
    
\end{frame}

\begin{frame}{GAMs: Addressing Autocorrelation}

    \textbf{Lack of Independence}
    \begin{itemize}
        \item Traditional GLMs assume all observations are strictly independent. For daily ridership, treating sequential days as independent ignores inherent time-series momentum, leading to severe residual autocorrelation.
        \item We correct this by embedding a 1-day lagged response ($\text{Trips}_{t-1}$) directly into the Negative Binomial structure, we explicitly model the temporal dependence:
    \end{itemize}
    
    \vspace{-0.2cm}
    \begin{equation*}
        \log(\mu_t) = \text{Environmental \& Calendar Predictors}_t + \phi(\text{Trips}_{t-1})
    \end{equation*}
    \vspace{-0.2cm}

\end{frame}

% --- SLIDE 1: THE BAYESIAN GAM MODEL ---
\begin{frame}{Bayesian GAM: Model Structure \& Results}

    \textbf{Model Formulation}
    \vspace{-0.2cm}
    \begin{equation*}
        \log(\mu_t) = \beta_0 + s(\text{Max Temp}_t) + \beta_1(\text{Precip}_t) + \boldsymbol{\gamma}(\text{Day Category}_t) + \epsilon_t
    \end{equation*}
    \vspace{-0.4cm}
    \begin{equation*}
        \epsilon_t = \rho \epsilon_{t-1} + \nu_t \quad \text{(AR1 Residual Autocorrelation)}
    \end{equation*}
    
    \vspace{0.2cm}
    
    \textbf{Posterior Estimates (Poisson Family)}
    \begin{table}
        \centering
        \small
        \begin{tabular}{lrlc}
            \toprule
            \textbf{Predictor} & \textbf{Estimate} & \textbf{Error} & \textbf{95\% Credible Interval} \\
            \midrule
            Intercept & $4.88$ & $0.04$ & $[4.81, 4.95]$ \\
            Precipitation (Hours) & $-0.07$ & $0.01$ & $[-0.09, -0.06]$ \\
            Saturday & $0.29$ & $0.07$ & $[0.16, 0.42]$ \\
            Sunday & $0.43$ & $0.06$ & $[0.31, 0.56]$ \\
            Statutory Holiday & $0.78$ & $0.13$ & $[0.53, 1.03]$ \\

    \end{table}

\end{frame}
